FUNCTION GET_DATE_DIFF(
    START_DATE                  IN DATE,
    END_DATE                    IN DATE,
    WEEKEND_DAYS                IN NUMBER
    )
RETURN NUMBER
IS 
  NUM_DAYS NUMBER;
  I NUMBER;
  WEEKEND_NUM NUMBER:=0;
BEGIN
  NUM_DAYS:= TRUNC(END_DATE) - TRUNC(START_DATE);
  --DBMS_OUTPUT.PUT_LINE(NUM_DAYS); 
  
  FOR I IN 0..NUM_DAYS
  LOOP
     IF TO_CHAR(START_DATE+I, 'DY', 'NLS_DATE_LANGUAGE=ENGLISH') IN ('SAT', 'SUN') THEN
      --DBMS_OUTPUT.PUT_LINE(NUM_DAYS); 
      WEEKEND_NUM := WEEKEND_NUM + 1;
     END IF;
  END LOOP;
  
  RETURN WEEKEND_NUM;

   
END GET_DATE_DIFF;
